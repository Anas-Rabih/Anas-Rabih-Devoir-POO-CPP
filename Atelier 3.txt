#include<iostream>
#include<string>
using namespace std;

class voiture{
    string marque;
    string modele;
    int annee;
    float kilometrage;
    float vitesse;
 public:
  void valeursParDefaut(){
    marque="inconnu" ;
    modele="inconnu";

}
